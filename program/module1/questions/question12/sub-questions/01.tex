Рассмотрим $\mathcal{L} = \mathcal{L}_1 \oplus \mathcal{L}_2$.

\subsection{
    Проекция вектора на подпространство вдоль другого подпространства.
}

\begin{definition}
    \textit{\textbf{Проекцией вектора $\vec{x}$ на подпространство $\mathcal{L}_1$ вдоль подпространства $\mathcal{L}_2$}} называется вектор $x_1$ из представления $\vec{x} = \vec{x}_1 + \vec{x}_2$, $\vec{x}_1 \in \mathcal{L}_1, \vec{x}_2 \in \mathcal{L}_2$.
\end{definition}
